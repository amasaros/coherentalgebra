\section*{Solving by Substitution}
Solve each system below. Each system is set up to make substitution the easiest method.
\begin{smenum}
\maitemxx
{\begin{System}
x=y-2\\
z=3y-5\\
3x+4y-4z=19
\end{System}}
{\begin{System}
z=4x+1\\
y=-2x+4\\
4x+y+z=-7
\end{System}}
\maitemxx
{\begin{System}
x=3y-2\\
y=5z-4\\
z=-5x+6
\end{System}}
{\begin{System}
x=3z-7\\
y=-x+3\\
z=-y+4
\end{System}}
\end{smenum}

\section*{Solving by Elimination}
