\section*{Polynomial Equations in Standard Form}
Solve by factoring. In these problems, if the polynomial has a prime quadratic factor, that factor is never zero.
\begin{smenum}
\mitemxx
{$x=3\wordor 4$}
{$x=0\wordor 4$}
\mitemxx
{$x=-3\wordor 3$}
{$x=0,-2,\wordor\frac{4}{3}$}
\mitemxx
{$x=-4,4,\wordor 9$}
{$x=0,-1,-\frac{1}{2},\wordor\frac{1}{2}$}
\mitemxx
{$x=-5\wordor 2$}
{$x=-1\wordor 4$}
\end{smenum}

\section*{Polynomial Equations in Any Form}
Solve by factoring. In these problems, if the polynomial has a prime quadratic factor, that factor is never zero.
\begin{smenum}
\mitemxx
{$x=-3,\frac{3}{2},\wordor 3$}
{$x=-9\wordor 8$}
\mitemxx
{$x=0,3,\wordor 5$}
{$x=-2\wordor 5$}
\end{smenum}
