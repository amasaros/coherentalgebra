\section{Perfect Squares}
For each trinomial below:
\begin{clist}
\item Determine whether two of its terms are perfect squares.
\item If so, determine whether the middle term makes it a perfect saure.
\item If it is a perfect square, factor it as such. (Don't factor if it isn't a perfect square.)
\end{clist}
\begin{senum}
\item\squares{$x^2=(x)^2$, $36y^2=(6y)^2$}{$2(x)(6y)=12xy$}{$(x+6y)^2$}
\item\squares{$25x^2=(5x)^2$, $100=10^2$}{$2(5x)(10)\neq 12x$}{\nos}
\item\squares{\nos}{N/A}{N/A}
\item\squares{$25z^2=(5z)^2$, $36=6^2$}{$-2(5z)(6)=-60z$}{$(5z-6)^2$}
\item\squares{$9q^2=(3q)^2$, $4=2^2$}{$2(3q)(2)\neq 38q$}{\nos}
\item\squares{$w^6v^2=(w^3v)^2$, $25w^2v^6=(5wv^3)^2$}{$2(w^3v)(5wv^3)=10w^4v^4$}{$(w^3v+wv^3)^2$}
\item\squares{$25=5^2$, $64x^2=(8x)^2$}{$-2(5)(8x)=-40x$}{$(5-8x)^2$}
\item\squares{$25x^2=(5x)^2$, $25=5^2$}{$2(5x)(5)\neq 42x$}{\nos}
\end{senum}
