\section*{Equations with Two or More Logarithms}
Solve each equation. Be sure to check for extraneous solutions.
\begin{smenum}
\maitemxx
{\log_3(x-3)+\log_3(x-5)=1}
{\log_6(x-5)=1-\log_6(x-6)}
\maitemxx
{\log_2(x-2)=3-\log_2(8-x)}
{\log_3(x+5)-\log_3(x-3)=2}
\maitemxx
{\log_5(x-3)=\log_5(x-5)+1}
{\log_4(x-2)=\log_4(x+4)-1}
\end{smenum}
