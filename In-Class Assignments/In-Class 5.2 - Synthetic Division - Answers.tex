\section{Synthetic Division}
Use synthetic division to divide. Write the result as quotient and remainder.
\begin{smenum}
\mitemxx
{\twoform{5x^3-4x^2+5x+3}{20}{x-3}}
{\twoform{4x^2-5x+2}{5}{x+2}}
\mitemxx
{$-7x^2-2x-1$}
{\twoform{5x^5+x^4-3x^3+7x^2+3x-1}{1}{x-2}}
\mitemxx
{\twoform{4x^4+4x^3+3x^2+17x+42}{133}{x-3}}
{\twoform{9x^5-17x^4+26x^3-47x^2+98x-189}{373}{x+2}}
\end{smenum}
In these exercises, be careful of missing terms.
\begin{smenum}
\mitemxx
{\twoform{4x^3+2x^2+10x+10}{1}{x-1}}
{\twoform{-x^2+2x-11}{14}{x+2}}
\mitemxx
{\twoform{9x^4+9x^3+6x^2+8x+8}{12}{x-1}}
{$x^4+2x^3+4x^2+8x+16$}
\end{smenum}

\section{Synthetic Division to Evaluate}
Let $P(x)=2x^4-11x^3+11x^2+15x-9$. Use synthetic division to evaluate each of the following. (You may check your work on your calculator, but use synthetic division first.)
\begin{smenum}
\mitemxxxx
{$P(3)=0$}{$P(1)=8$}{$P(-2)=125$}{$P\left(\frac{1}{2}\right)=0$}
\mitemx
{$3$ and $\frac{1}{2}$}
\end{smenum}

\section{Factors and Roots}
For each list of numbers, give a polynomial with the given list of roots, each with multiplicity 1, and given leading coefficient.
\begin{smenum}
\mitemxx
{$4(x-1)(x+3)(x-2)(x-5)$}
{$-2x(x+\frac{1}{3})(x-2)(x+\sqrt{3})$}

Give a list of numbers with the given roots, each with the given multiplicity, and a leading coefficient of 1.
\begin{smenum}
\mitemxx
{$x(x-1)^2(x+7)^3$}
{$(x+2)(x+5)^5(x-2)$}
\end{smenum}

\begin{smenum}
