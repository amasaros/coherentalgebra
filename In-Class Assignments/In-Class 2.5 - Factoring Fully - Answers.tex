\section*{Factoring Fully}
Factor each polynomial \textbf{fully}.  This means that if any factor is not prime, you should factor further.  Continue until all factors are prime.  If the polynomial is prime, write ``prime.''
\begin{smenum}
\mitemxxx
{$3ab(b+2a)$}
{$x(x-4)$}
{$mn^2p(15m+2mp-2n)$}
\mitemxxx
{$5ax(1-5a)$}
{$3x^2(3-y)(3+y)$}
{$m(m^2-5n)(m^2+5n)$}
\mitemxxx
{$xy(2z-7)(z+1)$}
{Prime}
{$3(m^2+3m+1)$}
\mitemxxx
{Prime}
{$4z(z-1)(z+1)(z-2)(z+2)$}

\end{smenum}
