\section*{Rational Functions}
For each pair of functions $f$ and $g$, write the function $\Gp{\frac{f}{g}}(x)$ as a rational function, and find its domain.  \textbf{Do not simplify} your rational function.
\begin{smenum}
\mitemxx
{$f(x)=2x^2-5x+3,\quad g(x)=2x^2-5x-7$}
{$f(x)=2x-3,\quad g(x)=4x^2-9$}
\end{smenum}
