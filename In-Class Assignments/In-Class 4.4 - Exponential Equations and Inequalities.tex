\section{Exponential Equations}
Solve each equation below.
\begin{clist}
\item Give the solution as an exact value, in terms of the common log if necessary.
\item If the answer in part (a) included a log, give a decimal approximation rounded to three decimal places.
\end{clist}
\begin{smenum}
\maitemxxxx
{3^x=7}
{4^{2x-3}=8}
{5\left(\frac{1}{3}\right)^x=28}
{3\cdot 2^{x-5}+17=104}
\maitemxxxx
{3^x=4^{x-2}}
{5^{2x-3}=4^{3x+1}}
{7^{2x+1}=2^{5x-2}}
{\left(\frac{2}{3}\right)^x=\left(\frac{5}{6}\right)^{x+10}}
\end{smenum}
Solve each equation below by pretending it's a polynomial.
\begin{clist}
\item Give the solution as an exact value, in terms of the common log if necessary.
\item If the answer in part (a) included a log, give a decimal approximation rounded to three decimal places.
\end{clist}
\begin{smenum}
\maitemxx
{2^{2x}-12\cdot 2^x+32=0}
{25^x+3\cdot 5^x=4}
\maitemxx
{2\cdot 3^{2x}-7\cdot 3^x+6=0}
{5^{2x-3}-6\cdot 5^{x-2}+1=0}
\end{smenum}

\section{Exponential Inequalities}
Solve each inequality below.
\begin{clist}
\item Give the solution in terms of an exact value, in terms of the common log if necessary.
\item If the answer in part (a) included a log, give the solution in terms of a decimal approximation rounded to three decimal places. \emph{Be sure to round in a way that respects the inequality.}
\end{clist}
\begin{smenum}
\maitemxxxx
{3^x>\frac{1}{81}}
{5^{x-2}\leq 25}
{1.2^x<2}
{0.9^x>0.1}
\maitemxx
{1500(1.05)^{2x-3}>3500}
{17\left(\frac{1}{2}\right)^{x/1.9}<1}
\end{smenum}
