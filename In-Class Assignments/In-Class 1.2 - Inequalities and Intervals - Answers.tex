\section*{Intervals in the Number Line}
\begin{senum}
\item \twopartans{$x<-4$}{$(-\infty,-4)$}
\item \twopartans{$-4\leq x<4$}{$[-4,\infty)$}
\item \twopartans{$0\leq x\leq 3$}{$[0,3]$}
\item \twopartans{$x>-2$}{$(-2,\infty)$}
\item \twopartans{$x\leq -3$}{$(-\infty,-3]$}
\item \twopartans{$x\geq 2$}{$[2,\infty)$}
\item \twopartans{$-5<x\leq -3$}{$(-5,-3]$}
\item \twopartans{$-2<x<5$}{$(-2,5)$}
\end{senum}
