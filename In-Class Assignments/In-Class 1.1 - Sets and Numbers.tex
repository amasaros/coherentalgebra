\section*{Set Builder Notation}
For each set $A$ given below, list three elements of the set. Then, write the given number $x$ in a form that shows that $x\in A$.
\begin{senum}
\item $A=\Set{\frac{n}{8}\mid n\in\Z}\qquad x=\frac{1}{2}$
\item $A=\Set{\frac{n}{d}\mid \mbox{$n$ is odd, $d$ is even}}\qquad x=0.75$
\item $A=\Set{x-2\mid x\in\N}\qquad x=-1$
\item $A=\Set{\frac{1}{a}\mid a\in\Q}\qquad x=3$
\end{senum}
