\section{End Behavior of a Polynomial}
Describe the end behavior of each polynomial below.
\begin{smenum}
\maitemxx
{x^8+7x^4-7x^3-8}
{-8x^6-2x^5+9x^4-x^2-2x}
\maitemxx
{-5x^7-2x^4+18x^2-1}
{3x^9-14x^2+7x-3}
\end{smenum}
For each polynomial below:
\begin{clist}
\item Compute the leading term
\item Analyze the end behavior
\end{clist}
\begin{smenum}
\maitemxx
{(x+6)^3(x-5)^3(3+x)}
{5(8-x)^2(3-x)^3(x-2)(4+3x)}
\maitemxx
{-7x^2(5+x)(7-x)^2(x-1)^3(x-6)}
{-4(2x+1)^2(3x-2)(5-x)^3}
\end{smenum}
