\begin{smenum}

\mitemx

\mitemx

\mitemx

\section*{Synthetic Division to Evaluate}
Let $P(x)=2x^4-11x^3+11x^2+15x-9$. Use synthetic division to evaluate each of the following. (You may check your work on your calculator, but use synthetic division first.)
\begin{smenum}
\mitemxxxx
{$P(3)$}{$P(1)$}{$P(-2)$}{$P\left(\frac{1}{2}\right)$}
\mitemx
{Which of the numbers you used above are roots of $P$?}
\end{smenum}

\section*{Factors and Roots}

\begin{smenum}
