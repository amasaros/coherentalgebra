\section*{Solving Rational Equations}
For each rational equation below:
\begin{clist}
\item List all values of the variable that will result in multiplying by 0 when clearing denominators.
\item Solve by clearing denominators.
\item List all extraneous solutions that had to be discarded, if any.
\end{clist}
\begin{smenum}
\mitemxxx
{$\dfrac{3}{2u}-\dfrac{1}{3}=\dfrac{5}{6u}$}
{$\dfrac{4}{3x-3}=\dfrac{7}{4x+1}$}
{$\dfrac{2x-1}{8}=\dfrac{x}{4}-\dfrac{x-1}{x-2}$}
\mitemxxx
{$\dfrac{4}{3x}-\dfrac{2}{x+4}=\dfrac{2}{3x+12}$}
{$\dfrac{3}{y^2-4}=\dfrac{-2}{5y+10}$}
{$\dfrac{y-2}{y+3}=\dfrac{3}{8}$}
\mitemxx
{$\dfrac{5}{h+5}-\dfrac{2}{h^2+2h-15}=\dfrac{2}{h-3}$}
{$\dfrac{3a}{a^2-2a-15}-\dfrac{a}{a+3}=\dfrac{2a}{a-5}$}
\mitemxx
{$\dfrac{1}{x-2}+\dfrac{9}{x+5}=3-\dfrac{7}{x^2+3x-10}$}
{$\dfrac{2x+5}{3x-2}=\dfrac{10-6x}{9x^2-30x+16}+\dfrac{1}{3x-8}$}
\mitemxx
{$\dfrac{x+1}{x^2+2x-24}-\dfrac{3}{x^2-x-12}=\dfrac{3}{x^2+9x+18}$}
{$\dfrac{p-5}{p+5}-2+\dfrac{p+15}{p-5}=\dfrac{50}{p^2-25}$}
\mitemxx
{$\dfrac{1}{y^2-5y+6}+\dfrac{y}{y^2-2y-3}=\dfrac{y+2}{y^2-y-2}$}
{$\dfrac{1}{b+2}-\dfrac{2}{b^2-1}=\dfrac{-2}{b^3+2b^2-b-2}$}
\end{smenum}
