\begin{smenum}
\mitemxx
{$x^2-x$}
{$18r^4-27r^6$}
\mitemxx
{$w^3v^2+w^2v+wv^2$}
{$21t^4u^3-105t^5u^3v+63t^6u^2v^4$}
\mitemxx
{$10g+4g^3-100g^4+120g^5$}
{$4a^3y^3-6ax^2y^2+4axy^2$}
\end{smenum}
For each polynomial, factor out a \textbf{negative} GCF.
\begin{smenum}
\mitemxx

\mitemxx
{$-6x^2-17x+24$}
{$-21a^4bc^3-15a^3b^3c^3+27a^2b^2c^4$}
\end{smenum}
