\section{Solutions to Radical Equations}
Decide whether each value is a solution to the given equation.
\begin{smenum}
\mitemxx
{$1+\sqrt{x}=\sqrt{2x+1}$; $x=4$}
{$\sqrt[3]{3x-4}=\sqrt[3]{5x+2}$; $x=-3$}
\mitemxx
{$\sqrt{x^2-3x+5}=\sqrt{x^2-7x+9}$; $x=2$}
{$\sqrt[3]{2x+4}+\sqrt{1-5x}=\sqrt[4]{x^2-3x+6}$; $x=-2$}
\end{smenum}

\section{Equations with an Isolated Radical}
Solve.  Be sure to eliminate any extraneous solutions. If there is no solution, write ``no solution.''
\begin{smenum}
\mitemxxx
{$\sqrt[3]{x}=-2$}
{$\sqrt{x+3}=5$}
{$\sqrt[4]{x+7}=-1$}
\mitemxxx
{$\sqrt{x^2-4}=x+2$}
{$\sqrt{21+4x}=2x+3$}
{$\sqrt[3]{y^2+y+7}=3$}
\end{smenum}

\section{Equations with One Radical}
Solve.  Be sure to eliminate any extraneous solutions. If there is no solution, write ``no solution.''
\begin{smenum}
\mitemxxx
{$\sqrt{y-6}+2=9$}
{$\sqrt[3]{n+3}-2=-4$}
{$2\sqrt[4]{m+3}+4=2$}
\mitemxxx
{$x-\sqrt{x}=6$}
{$\sqrt{6v+1}-1=v$}
{$\sqrt{x^2+4}-x=4$}
\mitemxx
{$x-\sqrt[3]{x^3-7}=1$}
{$\sqrt[3]{5x^2+28x+31}-x=3$}
\end{smenum}

\section{Equations with Two Radicals}
Solve.  Be sure to eliminate any extraneous solutions. If there is no solution, write ``no solution.''
\begin{smenum}
\mitemxx
{$\sqrt{m+2}=\sqrt{2m-3}$}
{$2\sqrt[4]{3h-4}-3\sqrt[4]{h+4}=0$}
\mitemxx
{$2\sqrt{x+6}-\sqrt{4x+24}=0$}
{$\sqrt{4x-5}=\sqrt{6x+5}$}
\mitemxx
{$\sqrt{-5x+56}=\sqrt{-x+44}-2$}
{$\sqrt{24w-95}-2\sqrt{5w-20}=1$}
\mitemxx
{$\sqrt{-5d+41}=3+\sqrt{-11d+56}$}
{$3\sqrt{-5w+11}=\sqrt{-33w+67}-2$}
\end{smenum}


\section{Inverse of a Radical Function}
Give the inverse of each radical function.  If it is an even root, be sure to include any required restrictions on the domain.
\begin{smenum}
\mitemxxxx
{$f(x)=\sqrt[3]{x}$}
{$g(x)=\sqrt[6]{x}$}
{$h(x)=-\sqrt{x}$}
{$k(x)=\sqrt[5]{x-3}+2$}
\mitemxxx
{$s(x)=\frac{1}{2}\sqrt{2x-5}-7$}
{$t(x)=-2\sqrt[3]{7-x}+4$}
{$v(x)=\sqrt[42]{3x-5}$}
\end{smenum}

\section{Inverses of Power Functions}
Give the inverse of each power function.  Note the domain restrictions.
\begin{smenum}
\mitemxxx
{$f(x)=x^3$}
{$g(x)=x^6,~x\geq 0$}
{$h(x)=x^4,~x\leq 0$}
\mitemxxx
{$k(x)=x^3-5$}
{$s(x)=(x-2)^7$}
{$t(x)=(4-x)^2,~4-x\geq 0$}
\end{smenum}
