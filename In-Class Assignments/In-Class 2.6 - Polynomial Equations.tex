\section{Polynomial Equations in Factored Form}
Solve.
\begin{smenum}
\mitemxx
{$(x-4)(x-2)(x+1)=0$}
{$4x(x-3)=0$}
\mitemxx
{$(5x-3)(2x+1)(x+4)=0$}
{$x^2(2x+5)(3x-2)(x-3)^2=0$}
\mitemxx
{$x(x-6)(4-x)(5-3x)=0$}
{$x(2x+1)(5x-2)(5-2x)=0$}
\end{smenum}

\section{Polynomial Equations in Standard Form}
Solve by factoring. In these problems, if the polynomial has a prime quadratic factor, that factor is never zero.
\begin{smenum}
\mitemxx
{$x^2-7x+12=0$}
{$x^2-4x=0$}
\mitemxx
{$x^4-5x^2-36=0$}
{$3x^3+2x^2-8x=0$}
\mitemxx
{$x^3-9x^2-16x+144=0$}
{$4x^5+4x^4-x^3-x^2=0$}
\mitemxx
{$x^4+6x^3+9x^2-100=0$}
{$x^4-9x^2-24x-16=0$}
\end{smenum}

\section{Polynomial Equations in Any Form}
Solve by factoring. In these problems, if the polynomial has a prime quadratic factor, that factor is never zero.
\begin{smenum}
\mitemxx
{$2x^3-18x=3x^2-27$}
{$(x-3)(x+4)=60$}
\mitemxx
{$x^3-7x^2+21x+9=(x+3)^2$}
{$(2x+5)(x-2)=x(x+4)$}
\end{smenum}
