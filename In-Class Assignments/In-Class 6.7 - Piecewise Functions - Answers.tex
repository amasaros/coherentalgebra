\section*{Building a Piecewise Function}
For each list of definitions below, combine the three parts into a single piecewise function. Then, state the domain of the piecewise function.
\begin{smenum}
\mitemxx
{$f(x)=\begin{pw}%
2x-4&2<x<4\\%
x^2-1&4\leq x<6\\%
x-2&6\leq x<8\end{pw}$\\%
$\dom f=\lrint{2}{8}$}
{$g(x)=\begin{pw}%
2x-3&x<2\\%
5x-2&x>4\end{pw}$\\%
$\dom{g}=\rint{2}\cup\lint{5}$}
\mitemx
{$h(x)=\begin{pw}%
x&x\neq 5\\%
-x&x=5\end{pw}$\\$\dom h=\R$}
\end{smenum}

\section*{Evaluating a Piecewise Function}
Evaluate each function below for the given values. Write ``undefined'' if the given value is not in the domain.
\begin{senum}
\item \begin{nmenum}
\mitemxxxx
{$f(-7)=-6$}
{$f(-3)=-6$}
{$f(2)$ is undefined}
{$f(0)=3$}
\end{nmenum}
\item \begin{nmenum}
\mitemxxxx
{$g(-2)=2$}
{$g(12)=23$}
{$g(4)$ is undefined}
{$g(5)$ is undefined}
\end{nmenum}
\item \begin{nmenum}
\mitemxxxx
{$h(-1.4)=-1$}
{$h(0.5)=-1$}
{$h(2)=2$}
{$h(2.5)$ is undefined}
\end{nmenum}
\end{senum}
