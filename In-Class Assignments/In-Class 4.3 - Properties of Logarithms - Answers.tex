You may wish to reference this list of log rules while you work on this assignment. You should plan to memorize these in the long run.

\[
\log_b(xy)=\log_b x+\log_b y
\qquad
\log_b\left(\frac{x}{y}\right)=\log_b x - \log_b y
\qquad
\log_b\left(x^p\right)=p\log_b x
\qquad
\log_b x=\frac{\log_c x}{\log_c b}
\]

\section{Rewriting with change of base}
Rewrite each expression in terms of the common logarithm.
\begin{smenum}
\maitemxx
{\log_5 x=\dfrac{\log x}{\log 5}}
{\log_3 4=\dfrac{\log 4}{\log 3}}
\maitemxx
{\log_{1/2} (x-3)=\dfrac{\log(x-3)}{\log\frac{1}{2}}}
{\log_8 (x^2+1)=\dfrac{\log(x^2+1)}{\log 8}}
\end{smenum}
For each expression below:
\begin{clist}
\item Rewrite the expression as a single logarithm.
\item Evaluate the expression on your calculator. 
\item Check by rewriting the single logarithm in exponential form.
\end{clist}
\begin{smenum}
\mitemxx
{$\dfrac{\log 4}{\log 2}=\log_2 4=2$\\$2^2=4$}
{$\dfrac{\log 25}{\log \frac{1}{5}}=\log_{1/5} 25=-2$\\$\Gp{\frac{1}{5}}^{-2}=25$}
\mitemxx
{$\dfrac{\log 2}{\log 16}=\log_{16} 2=\frac{1}{4}$\\$16^{1/4}=2$}
{$\dfrac{\log 0.25}{\log 0.5}=\log_{0.5} 0.25=2$\\$\Gp{0.5}^2=0.25$}
\end{smenum}
