\section{Dividing by a Linear Binomial}
Divide.  In the exercises below, you should not have a remainder.
\begin{smenum}
\mitemxx
{$\Gp{y^2+4y+4}\div\Gp{y+2}$}
{$\Gp{15 x^3-56 x^2+85 x-84}\div\Gp{3x-7}$}
\mitemxx
{$\Gp{2 x^3+5 x^2+3 x+18}\div\Gp{2x+6}$}
{$\Gp{r^5-4 r^4+9 r^3-21 r^2+25 r-6}\div\Gp{r-2}$}
\end{smenum}

\section{Dividing by a Linear Binomial with Remainder}
Divide.  In the exercises below, you may have a remainder.
\begin{smenum}
\mitemxx
{$\Gp{6x^2+7x+7}\div\Gp{3x-1}$}
{$\Gp{b^3-6b^2+14b-3}\div\Gp{b-3}$}
\mitemxx
{$\Gp{6 t^4-14 t^3+17 t^2+3 t-18}\div\Gp{3t-4}$}
{$\Gp{5 t^6-6 t^5-9 t^4-14 t^3-23 t^2-4 t+1}\div\Gp{t+1}$}
\end{smenum}

\section{Dividing by a Polynomial}
Divide.  In the exercises below, you may have a remainder.
\begin{smenum}
\mitemxx
{$\Gp{t^4+t^3-2 t^2-3 t-1}\div\Gp{t^2+2t+1}$}
{$\Gp{x^3+4 x^2-20 x-6}\div\Gp{x^2-3x-2}$}
\mitemx
{$\Gp{x^4+4 x^3-21 x^2-20 x+6}\div\Gp{2x^2-4x+6}$}
\mitemx
{$\Gp{3 v^4-36 v^3+25 v^2-12 v-18}\div\Gp{v^3-11v^2+4v+2}$}
\end{smenum}


\section{Division with Missing Terms}
Divide.  The polynomials below are not given in descending order, and may be missing terms.  You will need to rearrange them and fill in missing terms appropriately in order to divide.  You may have a remainder.
\begin{smenum}
\mitemxx
{$\Gp{u^5+1}\div\Gp{u+1}$}
{$\Gp{3 x^4-5 x^3+15 x^2-10 x-8}\div\Gp{x^2+4}$}
\mitemxx
{$\Gp{x^5+2 x^4+2 x^3-3 x^2-6 x-6}\div\Gp{x^3-3}$}
{$\Gp{2 x^5+7 x^3-x^2-4 x-4}\div\Gp{2x^3-x-1}$}
\end{smenum}
