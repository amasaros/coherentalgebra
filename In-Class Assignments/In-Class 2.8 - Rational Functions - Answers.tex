\section*{Rational Functions}
For each pair of functions $f$ and $g$, write the function $\Gp{\frac{f}{g}}(x)$ as a rational function, and find its domain.  \textbf{Do not simplify} your rational function.
\begin{smenum}
\mitemx
{$\Gp{\frac{f}{g}}(x)=\dfrac{2x^2-5x+3}{2x^2-5x-7}$\qquad$\dom\frac{f}{g}=\Set{x\mid x\neq-1,\frac{7}{2}}$}
\mitemx
{$\Gp{\frac{f}{g}}(x)=\dfrac{2x-3}{4x^2-9}$\qquad$\dom\frac{f}{g}=\Set{x\mid x\neq-\frac{3}{2},\frac{3}{2}}$}
\end{smenum}

\section*{Evaluating Rational Functions}
Let $h(x)=\dfrac{2x^2-5x+3}{x^3-3x^2+3x-3}$.  Find
\begin{smenum}
\mitemxxxx
{$h(2)=-1$}
{$h(-3)=-\frac{6}{11}$}
{$h(\frac{1}{2})=-\frac{8}{17}$}
{$h(1)$ is undefined}
\end{smenum}
Let $s(x)=\dfrac{x^2-5x-11}{2x^2+5}$
\begin{smenum}
\mitemxxxx
{$s(7)=\frac{3}{103}$}
{$s(-1)=-\frac{5}{7}$}
{$s(1.5)=-\frac{65}{38}$}
{$s(0)=-\frac{11}{5}$}
\end{smenum}
