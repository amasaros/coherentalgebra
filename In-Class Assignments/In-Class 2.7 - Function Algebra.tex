\section*{Combined functions}
Define the following functions:
\[
\begin{array}{r@{\;=\big\{(}r@{,}r@{),(}r@{,}r@{),(}r@{,}r@{),(}r@{,}r@{),(}r@{,}r@{)\big\}}}
f&-7&-9&-1&-3&1&3&5&0&9&-8\\
g&-7&-4&1&-8&5&1&7&0&9&-7\\
h&-7&0&-1&-5&1&-7&7&-1&9&-5
\end{array}
\]
Find each of the following.  If the given value is not in the domain of the given function, write ``undefined.''
\begin{smenum}
\mitemxxxx
{$\Gp{f\cdot g}(9)$}
{$\Gp{f-h}(5)$}
{$\Gp{h-g}(7)$}
{$\Gp{g\cdot h}(-7)$}
\mitemxxxx
{$\Gp{f+g}(-7)$}
{$\Gp{g-f}(1)$}
{$\Gp{\dfrac{g}{h}}(-7)$}
{$\Gp{\dfrac{f}{g}}(5)$}
\mitemxxxx
{$\Gp{h\cdot f}(7)$}
{$\Gp{g+h}(7)$}
{$\Gp{\dfrac{h}{f}}(-1)$}
{$\Gp{h+f}(-1)$}
\end{smenum}
Find each domain.
\begin{smenum}
\mitemxxxo
{$\dom f$}
{$\dom g$}
{$\dom h$}
\mitemxxxx
{$\dom\Gp{g+h}$}
{$\dom\Gp{h-f}$}
{$\dom\Gp{f\cdot g}$}
{$\dom\Gp{\dfrac{h}{g}}$}
\end{smenum}
