\section*{Solving by Elimination}
Solve each system by elimination. If the system is inconsistent, write ``No Solution.'' If the system is dependent, write ``Infinitely many solutions,'' and describe the solutions in terms of one variable.
\begin{smenum}
\maitemxx
{\threesln{-2,-2,3}}
{\threesln{1,4,1}}
\mitemxx
{\threesln{1,-3,-3}}
{No solution}
\mitemxx
{\threesln{1,0,3}}
{Infinitely many solutions\\
In terms of $x$: $\Gp{x,\frac{1}{2}x-1,\frac{3}{2}x-6}$\\
In terms of $y$: $\Gp{2y+2,y,3y-3}$\\
In terms of $z$: $\Gp{\frac{2}{3}z+4,\frac{1}{3}z+1,z}$}
\mitemxx
{\threesln{2,1,1}}
{Infinitely many solutions\\
In terms of $x$: $\Gp{x,x-3,-2x+1}$\\
In terms of $y$: $\Gp{y+3,y,-2y-5}$\\
In terms of $z$: $\Gp{-\frac{1}{2}z+\frac{1}{2},-\frac{1}{2}z-\frac{5}{2},z}$}
\end{smenum}
