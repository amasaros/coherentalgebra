\subsection{Power Functions: Whole Number Powers} 
\functionfamily{Power Functions (even powers)}{%
	\foreach \p in {2} \draw[graph,domain={-4^(1/\p)}:{4^(1/\p)}] plot (\x,{(\x)^\p});
}
\functionfamily{Power Functions (odd powers)}{%
	\foreach \p in {3} \draw[graph,domain={-4^(1/\p)}:{4^(1/\p)}] plot (\x,{(\x)^\p});
}
\begin{note}
If the power is larger, the graph becomes \makeblank[1in] near zero and \makeblank[1in] far from zero.
\end{note}
\begin{note}
Because we can \makeblank, every quadratic function is a power function. This is not true of all \makeblank[1.5in].
\end{note}
\begin{note}
Technically, a \makeblank[1in] function is a power function with exponent \makeblanksmall.
\end{note}
