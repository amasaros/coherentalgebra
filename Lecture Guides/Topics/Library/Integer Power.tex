\subsection{Power Functions: Negative Integer Powers} 
\functionfamily{Power Functions (odd powers)}{%
	\draw[asymp] (-4,0) -- (4,0);
	\draw[asymp] (0,-4) -- (0,4);
	\draw[graph,domain=0.25:4,samples=100] plot (\x,1/\x);
	\draw[graph,domain=0.25:4,samples=100] plot (-\x,-1/\x);
}
\functionfamily{Power Functions (even powers)}{%
	\draw[asymp] (-4,0) -- (4,0);
	\draw[asymp] (0,-4) -- (0,4);
	\draw[graph,domain=0.5:4,samples=100] plot (\x,1/\x/\x);
	\draw[graph,domain=0.5:4,samples=100] plot (-\x,1/\x/\x);
}
\begin{Note}
We can think of $F(x)=x^{-p}$ as $(f\circ g)(x)$, where $f(x)=\makeblank[1in]$ and $g(x)=\makeblank[1in]$. Most of these properties can be derived that way.
\end{Note}
