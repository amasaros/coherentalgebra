\subsection{Power Functions: Radicals} 
\begin{Note}
A radical function is a power function, because $\sqrt[p]{x}=\makeblank[1in]$.
\end{Note}
\functionfamily{Power Functions (even roots)}{%
	\draw[graphend,->,domain=0:2] plot (\x^2,\x);
	\fill[graphend] (0,0) circle (0.1);
}
\functionfamily{Power Functions (odd roots)}{%
		\draw[graph,domain={-4^(1/3)}:{4^(1/3)}] plot (\x^3,\x);
}
\begin{Note}
We can derive the rest of the rational power functions by $F(x)=x^{m/n}$ as $(f\circ g)(x)$, where $f(x)=\makeblank[1in]$ and $g(x)=\makeblank[1in]$.
\end{Note}
