\subsection{Symmetry} 
\begin{definition}[Symmetry]
A figure is \term{symmetric} under a transformation if applying the transformation leaves the figure the same.
\end{definition}
\begin{definition}[Even Functions]
A function $f(x)$ is called \term{even} if:
\begin{itemize}
\item its graph symmetric under reflection across the $y$-axis.
\item $f(-x)=f(x)$ for all $x$.
\end{itemize}
\end{definition}
\begin{definition}[Odd Functions]
A function $f(x)$ is called \term{odd} if:
\begin{itemize}
\item its graph symmetric under rotation by 180\degree\ around $(0,0)$.
\item $f(-x)=-f(x)$ for all $x$.
\end{itemize}
\end{definition}
\begin{note}
If a function were symmetric across the $x$-axis, we would have $\makeblanksmall=\makeblanksmall$, which only happens if $f(x)=\makeblanksmall$ for all $x$.
\end{note}
