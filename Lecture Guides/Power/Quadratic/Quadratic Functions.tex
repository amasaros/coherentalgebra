\subsection{Quadratic Functions} 
\begin{definition}[Quadratic Function]
A \term{quadratic function} is a polynomial function of degree 2 (that is, a polynomial function in which the polynomial is quadratic).
\end{definition}
\begin{example}
Determine whether each function is quadratic.
\begin{enumerate}\setabc
\item $f(x)=x^2-5x+3$\makeblanknewf
\item $g(x)=x^2-7$\makeblanknewf
\item $h(x)=(x-5)^2+2$\makeblanknewf
\item $k(x)=\sqrt{4x+3}$\makeblanknewf
\item $p(x)=x^3-4x^2+2x-1$\makeblanknewf
\item $q(x)=(2x-5)^4-1$\makeblanknewf
\end{enumerate}
\end{example}
