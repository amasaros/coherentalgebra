\subsection{The Vertex Formula} 
Like solving quadratic equations, we can go through the process of writing in vertex form just once, and record the resulting formula.
\begin{lpic}{}{15}
\foreach \y/\lbl in {1/Start in standard form,
										2/Factor out $a$,
										13/Compare vertex form}
	\regtextleft{0.25}{-\y}{\lbl};
\foreach \y/\lbl in {3/Complete the square,
										7/Distribute,
										9/Simplify}
	\fractextleft{0.25}{-\y}{\lbl};
\foreach \y/\lbl in {1/ax^2+bx+c,
										13/a(x-h)^2+k}
	\regtextleft{7}{-\y}{$f(x)=\lbl$};
\draw (6.5,0) -- (6.5,-15);
\begin{scope}[xshift=2*\value{cols}*\linespace-4*\linespace]
	\draw (-0.5,0) -- (-0.5,-15);
	\draw (1,-3) rectangle (4,-6);
	\draw (1,-4.75) -- ++(3,0);
	\draw (2.75,-3) -- ++(0,-3);
\end{scope}
\end{lpic}
\begin{fact}[The Vertex Formula]
For a quadratic function $f(x)=ax^2+bx+c$, coordinates of the vertex of the graph are given by
$$(h,k)=\left(\dfrac{-b}{2a},\dfrac{-b^2+4ac}{4a}\right)$$
\end{fact}
\begin{note}If you prefer, you can just remember $h=\frac{-b}{2a}$ and $f(h)=k$.\end{note}
