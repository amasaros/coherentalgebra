\subsection{Using the Quadratic Formula} 
\begin{procedure}{Solving a Quadratic Equation using the Quadratic Formula}
\begin{enumerate*}
\item Write the equation in standard form $ax^2+bx+c=0$
\item Identify the values of $a$, $b$, and $c$.
\item Plug into the formula.
\item If needed, simplify or get a decimal approximation for the solutions.
\end{enumerate*}
\end{procedure}
