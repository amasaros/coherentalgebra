%Adjust the horizontal margins
%How many columns will fit?
\setcounter{cols}{1 * \ratio{0.5\paperwidth - \lrmargin}{\linespace}}
\setlength{\halftext}{\value{cols}\linespace}
\setlength{\hmargtotal}{\paperwidth - 2\halftext}
%\setlength{\lrmargin}{0.5\hmargtotal}
\setlength{\lmargin}{\lrmargin}
\setlength{\rmargin}{\lrmargin}
%Adjust the bottom margin
%No text after the last horizontal line
\setcounter{rows}{1 * \ratio{\paperheight - \tmargin - \bmargin}{\linespace}}
\setlength{\bmargin}{\paperheight - \tmargin - \value{rows}\linespace + .25\linespace}
