\subsection{Examples from Pairs} 
\begin{Example}
Define the following functions:
\[
\begin{array}{r@{\;=\big\{(}r@{,}r@{),\;(}r@{,}r@{),\;(}r@{,}r@{),\;(}r@{,}r@{),\;(}r@{,}r@{)\big\}}}
f&-8&8&-6&-3&3&0&4&3&9&-3\\
g&-8&-6&-4&5&3&-7&7&-3&9&0\\
h&-8&3&0&-9&1&-8&3&-3&9&7\\
\end{array}
\]
Find each of the following.
\begin{enumerate}\setabc
\item\makeblankfill{$(f+g)(3)$ }
\item\makeblankfill{$(h-g)(1)$ }
\item\makeblankfill{$(f\cdot g)(9)$ }
\item\makeblankfill{$\left(\frac{g}{h}\right)(-8)$ }
\item\makeblankfill{$\left(\frac{f}{h}\right)(3)$ }
\end{enumerate}
\end{Example}
\begin{Example}
Define the following functions:
\[
\begin{array}{r@{\;=\big\{(}r@{,}r@{),\;(}r@{,}r@{),\;(}r@{,}r@{),\;(}r@{,}r@{),\;(}r@{,}r@{)\big\}}}
f&-8&8&-6&-3&3&0&4&3&9&-3\\
g&-8&-6&-4&5&3&-7&7&-3&9&0\\
h&-8&3&0&-9&1&-8&3&-3&9&7\\
\end{array}
\]
Find each of the following.
\begin{enumerate}\setabc
\item\makeblankfill{$\dom f$ }
\item\makeblankfill{$\dom g$ }
\item\makeblankfill{$\dom h$ }
\item\makeblankfill{$\dom(f+g)$ }
\item\makeblankfill{$\dom(f-h)$ }
\item\makeblankfill{$\dom(g\cdot h)$ }
\item\makeblankfill{$\dom\left(\frac{h}{g}\right)$ }
\end{enumerate}
\end{Example}
