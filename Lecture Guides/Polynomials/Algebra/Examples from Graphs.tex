\subsection{Examples from Graphs} 
\begin{Example}
The functions $f$, $g$, and $h$ are graphed below.
\begin{center}
\begin{tikzpicture}[x=0.5\linespace,y=0.5\linespace]
\useasboundingbox(-1/3*\linewidth-3*\linespace,-2.75*\linespace) rectangle 
								(1/3*\linewidth+3*\linespace,3*\linespace);
\setgraphsquare{5}
\begin{scope}[xshift=-1/3*\linewidth]
\GraphSmall
\draw[domain=0:4,samples=40,graph,->] plot (\x+1,{sqrt(\x)+1});
\draw[domain=0:4,samples=40,graph,->] plot (-\x-1,{sqrt(\x)+1});
\foreach \x in {-1,1} \draw[dotfull] (\x,1) circle (\dotsize);
\draw (-4,4) node[anchor=base] {$f$};
\end{scope}
\begin{scope}[xshift=0]
\GraphSmall
\draw[domain=-5:5,samples=80,graph] plot (\x,\x*\x/4-4);
\draw (-4,4) node[anchor=base] {$g$};
\end{scope}
\begin{scope}[xshift=1/3*\linewidth]
\GraphSmall
\draw[graphend] (-4,3) -- (1,-2);
\draw[graphend] (1,-3) -- (5,5);
\foreach \x/\y in {-4/3,1/-2,1/-3,5/5} \draw[dotempty] (\x,\y) circle (\dotsize);
\draw (-4,4) node[anchor=base] {$h$};
\end{scope}
\end{tikzpicture}
\end{center}
Find each of the following:
\begin{enumerate}\setabc
\item\makeblankfill{$(f+g)(2)$ }
\item\makeblankfill{$(g-h)(4)$ }
\item\makeblankfill{$(f\cdot h)(1)$ }
\item\makeblankfill{$\left(\frac{f}{h}\right)(-2)$ }
\item\makeblankfill{$\left(\frac{h}{g}\right)(4)$ }
\end{enumerate}
\end{Example}
