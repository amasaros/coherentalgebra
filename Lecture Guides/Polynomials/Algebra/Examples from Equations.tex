\subsection{Examples from Equations} 
\begin{Example}
Define the following functions.\\
\begin{tabularx}{\linewidth}{CCC}
$f(x)=2x-3$
&
$g(x)=x^2-3x-10$
&
$h(x)=x^2-9$
\end{tabularx}
Find each of the following.
\begin{enumerate}\setabc
\item\makeblankfill{$(f+g)(2)$ }
\item\makeblankfill{$(f-h)(-1)$ }
\item\makeblankfill{$(g\cdot h)(0)$ }
\item\makeblankfill{$\left(\frac{h}{g}\right)(5)$ }
\item\makeblankfill{$\left(\frac{f}{g}\right)(-4)$ }
\end{enumerate}
\end{Example}

\begin{lpic}{}{8}

\end{lpic}
Write an expression for the following. Do not simplify.
\begin{lpic}{}{2}
	\fractextright{1}{-1}{d. };
	\fractextleft{1}{-1}{$\left(\dfrac{g}{h}\right)(x)={}$};
\end{lpic}
\end{Example}
