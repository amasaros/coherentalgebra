\subsection{Polynomial Equations} 
\begin{definition}[Polynomial Equation]
An equation is a \term{polynomial equation} if the expressions on either side are polynomials, at least one of which is degree 2 or higher.
\end{definition}
\begin{Example}
$x^2-5x-14=0$ is a polynomial equation. Determine whether each value below is a solution to this equation:
\begin{enumerate}\setabc
\item\makeblankfill{$x=7$ }
\item\makeblankfill{$x=2$ }
\item\makeblankfill{$x=-2$ }
\end{enumerate}
\end{Example}
\begin{Example}
$(x-5)(x+3)=9$ is a polynomial equation. Determine whether each value below is a solution to the equation:
\begin{enumerate}\setabc
\item\makeblankfill{$x=6$ }
\item\makeblankfill{$x=3$ }
\item\makeblankfill{$x=-4$ }
\end{enumerate}
\end{Example}
\noi Note that polynomial equations typically have \makeblank solution.
