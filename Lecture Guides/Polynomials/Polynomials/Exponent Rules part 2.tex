\subsection{Exponent Rules (Power of a Product, Quotient, and Power)} 
\begin{theorem}[Power of a Power Rule]
For any real number $x$ and any positive integers $n$ and $p$,

\nic{$\left(x^n\right)^p=x^{np}$}
\end{theorem}
\begin{lpic}{}{7}
\regtextleft{0.5}{-1}{$\left(x^n\right)^p={}$};
\end{lpic}
\begin{theorem}[Power of a Product Rule]
For any real numbers $x$ and $y$ and any positive integer $n$,

\nic$\left(xy\right)^n=x^ny^n$
\end{theorem}
\begin{lpic}{}{4}
\regtextleft{0.5}{-1}{$\left(xy\right)^n={}$};
\end{lpic}
\begin{theorem}[Power of a Quotient Rule]
For any real numbers $x$ and $y\neq 0$ and any positive integer $n$,

\nic$\left(\dfrac{x}{y}\right)^n=\dfrac{x^n}{y^n}$
\end{theorem}
\begin{lpic}{}{7}
\fractextleft{0.5}{-1}{$\left(\dfrac{x}{y}\right)^n={}$};
\end{lpic}
