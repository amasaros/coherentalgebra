\subsection{Factoring Fully} 
\begin{definition}[Factoring Fully]
A polynomial is \term{fully factored} if:
\begin{itemize*}
\item The greatest common factor, if any, has been factored out.
\item All polynomial factors are prime.
\end{itemize*}
\end{definition}
\begin{example}
Fully factor the polynomial $x^4y^2-16y^2$
\begin{lpic}{}{9}
\foreach \x/\step in {-1/1. Factor out the GCF,
											-3/2. Factor the polynomial factor,
											-5/3. Factor each new polynomial factor,
											-8/4. Put it all together}
	\regtextleft{0}{\x}{\step};
\end{lpic}
\end{example}
