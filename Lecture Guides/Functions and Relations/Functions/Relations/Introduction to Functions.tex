\subsection{Introduction to Functions} 
\begin{definition}[Function]
A \term{function} is a special type of relation in which each element of the domain is paired with exactly one element of the range.
\end{definition}
\begin{Example}
Determine whether the relation $R$ is a function.
\[
R=\left\{(0,-1),(-9,8),(-1,1),(-9,5)\right\}
\]
\begin{itemize}
\item The pairs \blankpair\ and \blankpair\ have the same \makeblanksmall coordinate but different \makeblanksmall coordinates.
\item Therefore $R$ \makeblank[1in] a function.
\end{itemize}
\end{Example}
\begin{Example}
Determine whether the relation $T$ is a function.
\[
T=\left\{(2,0),(9,-1),(-4,5),(5,7)\right\}
\]
\begin{itemize}
\item We see that each \makeblanksmall only appears in \makeblank[1in] pair.
\item Therefore, $T$ \makeblank[1in] a function.
\end{itemize}
\end{Example}
\begin{Example}
Determine whether the relation $S$ is a function.
\[
S=\left\{(-6,-6),(-3,-6),(3,-6),(-2,4)\right\}
\]
\begin{itemize}
\item We see that each \makeblanksmall only appears in \makeblank[1in] pair.
\item Three of them have the same \makeblanksmall coordinate, but that doesn't matter in the definition of ``function.''
\item Therefore, $S$ \makeblank[1in] a function.
\end{itemize}
\end{Example}
