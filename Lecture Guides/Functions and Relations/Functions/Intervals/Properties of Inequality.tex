\subsection{Properties of Inequality} 
Inequalities have \emph{most} of the same properties as equality, with some important differences.
\begin{displaybox}[Anti-Symmetric Property of Inquality]
For any real numbers $a$ and $b$:
\[
\begin{array}{c@{\quad\mbox{if and only if}\quad}c@{\qquad}c@{\quad\mbox{if and only if}\quad}c}
a>b&b<a&
a\geq b&b\leq a
\end{array}
\]
Note that we can change the order, but we must \makeblank[1in] the inequality.
\end{displaybox}
\begin{displaybox}[Transitive Property of Inequality]
For any real numbers $a$, $b$, and $c$:
\[
\begin{array}{@{\mbox{If }}c@{\mbox{ and }}c@{\mbox{ then }}c@{\qquad}@{\mbox{If }}c@{\mbox{ and }}c@{\mbox{ then }}c}
a<b&b<c&a<c&
a>b&b>c&a>c\\
a\leq b&b\leq c&a\leq c&
a\geq b&b\geq c&a\geq c\\
\end{array}
\]
All inequalities must be in the same \makeblank[1.5in]. If we mix equality and not, the final result \makeblank[1in] include equality.
\end{displaybox}
\begin{displaybox}[Additive Property of Inequality]
For any real numbers $a$, $b$, and $c$:
\[
\begin{array}{c@{\quad\mbox{if and only if}\quad}c@{\qquad}c@{\quad\mbox{if and only if}\quad}c}
a<b&a+c<b+c&
a>b&a+c>b+c\\
a\leq b&a+c\leq b+c&
a\geq b&a+c\geq b+c
\end{array}
\]
This is \makeblank the additive property of equality.
\end{displaybox}
\begin{displaybox}[Multiplicative Property of Inequality]
For any real numbers $a$, $b$, and $c$, if $c>0$ (that is, $c$ is \makeblank[1in]):
\[
\begin{array}{c@{\quad\mbox{if and only if}\quad}c@{\qquad}c@{\quad\mbox{if and only if}\quad}c}
a<b&ac<bc&
a>b&ac>bc\\
a\leq b&ac\leq bc&
a\geq b&ac\geq bc
\end{array}
\]
If $c<0$ (that is, $c$ is \makeblank[1in]):
\[
\begin{array}{c@{\quad\mbox{if and only if}\quad}c@{\qquad}c@{\quad\mbox{if and only if}\quad}c}
a<b&ac>bc&
a>b&ac<bc\\
a\leq b&ac\geq bc&
a\geq b&ac\leq bc
\end{array}
\]
If we multiply or divide by a negatve number, we must \makeblank[1.5in] the inequality.
\end{displaybox}
