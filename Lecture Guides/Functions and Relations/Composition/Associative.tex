\subsection{Composition is Associative} 
\begin{fact}[Associative Property of Composition]
For any functions $f$, $g$, and $h$ (with appropriate domains and ranges),
$$f\circ(g\circ h)=(f\circ g)\circ h$$
We use this fact to justify writing $f\circ g\circ h$ without parentheses.
\end{fact}
\begin{Example}Let $f(x)=5x-2$, $g(x)=3x+1$, and $h(x)=\frac{1}{5}x+3$. Write expressions for each of
$$(f\circ(g\circ h))(x) \qquad \mbox{and} \qquad ((f\circ g)\circ h)(x)$$
Confirm that these are the same function.

\regtextleft{\value{fcols}+2}{-9}{$((f\circ g)\circ h)(x)={}$};
\end{lpicvsplit}
\end{Example}
