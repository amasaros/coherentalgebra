\subsection{Composition is Not Commutative} 
\begin{fact}[Composition is Not Commutative]
For functions $f$ and $g$ (with appropriate domains and ranges) in general
$$f\circ g\neq g\circ f$$
If this happens to be true for two functions, it is a coincidence.
\end{fact}
\begin{Example}The functions $f$ and $g$ are given in the table below.
\begin{center}
\begin{tabularx}{\linewidth}{|R|C|C|C|C|C|C|C|C|C|}
\hline
$x$&$-4$&$-3$&$-2$&$-1$&$0$&$1$&$2$&$3$&$4$\\\hline
$f(x)$&$2$&$1$&$4$&$3$&$6$&$5$&$8$&$7$&$10$\\\hline
$g(x)$&$-6$&$-4$&$-2$&$0$&$2$&$4$&$6$&$8$&$10$\\\hline
$h(x)$&$0$&$-1$&$0$&$1$&$0$&$-1$&$0$&$1$&$0$\\\hline
\end{tabularx}
\end{center}
$(f\circ g)(-1)=\makeblank[1.5in]$\hfill$(g\circ f)(-1)=\makeblank[1.5in]$\lbr
$(g\circ h)(3)=\makeblank[1.5in]$\hfill$(h\circ g)(3)=\makeblank[1.5in]$\lbr
$(f\circ h)(-4)=\makeblank[1.5in]$\hfill$(h\circ f)(-4)=\makeblank[1.5in]$
\end{Example}
\begin{Example}Let $p(x)=\frac{1}{2}x-5$ and $q(x)=4x+2$. Write expressions for each of
$$(p\circ q)(x) \qquad \mbox{and} \qquad (q\circ p)(x)$$
Confirm that these are different functions.
\begin{lpicvsplit}{2}{7}
\regtextleft{1}{-7}{$(p\circ q)(x)={}$};
\regtextleft{\value{fcols}+2}{-7}{$(q\circ p)(x)={}$};
\end{lpicvsplit}
\end{Example}
